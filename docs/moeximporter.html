131    def _MoexRequest(self, _type, _pparams = None, _params = None):
132        """Internal method for standartize https-queries to MOEX ISS.
133        
134        You don't need to call this method directly, but you may use it for
135        more specific queries.
136        
137        This method is wrapped into more direct methods to request data.
138        
139        Parameters
140        ----------
141        _type: MoexRequests
142            Determines the type of the request. Is used to generate an url with
143            required parameters. Algorithms to generate the url are implemented
144            in __init__ method of the class.
145        _pparams: dict, optional
146            Dictionary of pairs id:value where id is the string id of the
147            parameter and value is its string value. These parameters are
148            used in the url string and should be replace with values.
149            Example:
150            in the url template
151            https://iss.moex.com/iss/securities.json
152            parameter __SECCODE__ should be replaced with the actual security
153            ticker.  {'__SECCODE___':'GAZP'} is passed to the method.
154        _params: dict, optional
155            Dictionary of pairs id:value where id is a string id of the
156            parameter and value is its value. The value type depends on the
157            request and automatically converted for the url in this method.
158            Example:
159            request
160            https://iss.moex.com/iss/securities.json
161            may use 2 parameters in some cases: start and is_trading
162            https://iss.moex.com/iss/securities.json?start=0&is_trading=1
163            the argument should be {'start': 0, 'is_trading': '1'}
164                
165        Returns
166        -------
167        array_like
168            Returns the array of dictionaries from the MOEX ISS reply.
169            This array should be parsed and converted for usage.
170            More information you can find on https://iss.moex.com/iss/reference/
171        """
172        _res = None
173        _values = self.base_values
174        try:
175            _mr = self.requests_dictionary[_type]
176            _url = self.base_url + _mr['postfix']
177            _rpp = _mr['postfix_params']
178            if _pparams:
179                for _ppkey in _pparams:
180                    if _ppkey in _rpp:
181                        _url = _url.replace(_ppkey, _pparams[_ppkey]) 
182            _rp = _mr['params']
183            if _params:
184                for _pp in _params:
185                    if _pp in _rp:
186                        _values[_pp] = f'{_params[_pp]:{_rp[_pp]}}'
187            _data = urllib.parse.urlencode(_values)
188            _req = urllib.request.Request(_url+f'?{_data:s}', headers=self.base_header, method = self.method)
189            _resp = None
190            try:
191                _resp = urllib.request.urlopen(_req, context=ssl._create_unverified_context())
192            except urllib.error.HTTPError as e:
193                print('_MoexRequest(): HTTP Error ', e.code)
194            except urllib.error.URLError as e:
195                print('_MoexRequest(): Error ', e.reason)
196            else:
197                _res = json.load(_resp)
198            if _resp:
199                _resp.close()
200        except Exception as e:
201            print('MoexImporter::_MoexRequest(): ', e, file=sys.stderr)
202        return _res

280        """Internal method to request security list.
281        
282        Parameters
283        ----------
284        _is_trading: str
285            Select traded, non-traded or all securities:
286            ''  all securities,
287            '0' - non-traded securities,
288            '1' - traded securities.
289        _engine: str
290            Securities for the specific engine.
291        _market: str
292            Securities for the specific market.
293        _query: str
294            Search security the part of its name or ticker.
295
296        Returns
297        -------
298        array_like
299            List of securities.
300        """
301        _res = None
302        _params = {
303            'start': 0,
304            'is_trading': _is_trading,
305        }
306        _type = MoexRequests.GetSecuritiesAll
307        if _engine:
308            _type = MoexRequests.GetSecuritiesForEngine
309            _params['engine'] = _engine
310            if _market:
311                _type = MoexRequests.GetSecuritiesForMarket
312                _params['market'] = _market
313        if _query:
314            _type = MoexRequests.GetSecuritiesSearch
315            _params['q'] = _query
316        try:
317            isNext = True
318            while isNext:
319                isNext = False
320                _tmp = self._MoexRequest(
321                    _type,
322                    _params = _params
323                )
324                for _ti in _tmp:
325                    if 'securities' in _ti:
326                        if _ti['securities']:
327                            if not _res:
328                                _res = []
329                                
330                            _res += [
331                                {
332                                    _k:_sq[_k]
333                                    for _k in _sq
334                                    if _k in ['secid', 'shortname', 'name', 'regnumber', 'isin', 'is_traded', 'emitent_id', 'emitent_title', 'emitent_inn', 'gosreg', 'primary_boardid']
335                                } for _sq in _ti['securities']
336                            ]
337                            _params['start'] += self.limit
338                            if len(_ti['securities']) == self.limit:
339                                isNext = True
340                    
341        except Exception as e:
342            print('MoexImporter::_getSecurities(): ', e, file=sys.stderr)
343        return _res
