159    def _MoexRequest(self, _type, _pparams = None, _params = None):
160        """Internal method for standartize https-queries to MOEX ISS.  
161        
162        You don't need to call this method directly, but you may use it for
163        more specific queries.  
164        
165        This method is wrapped into more direct methods to request data.
166        
167        Parameters
168        ----------
169        _type: _MoexRequests
170            Determines the type of the request. Is used to generate an url with
171            required parameters. Algorithms to generate the url are implemented
172            in __init__ method of the class.
173        _pparams: dict, optional
174            Dictionary of pairs id:value where id is the string id of the
175            parameter and value is its string value. These parameters are
176            used in the url string and should be replace with values.  
177            Example:  
178            in the url template  
179            https://iss.moex.com/iss/securities.json  
180            parameter __SECCODE__ should be replaced with the actual security
181            ticker.  {'__SECCODE___':'GAZP'} is passed to the method.
182        _params: dict, optional
183            Dictionary of pairs id:value where id is a string id of the
184            parameter and value is its value. The value type depends on the
185            request and automatically converted for the url in this method.  
186            Example:  
187            request  
188            https://iss.moex.com/iss/securities.json  
189            may use 2 parameters in some cases: start and is_trading  
190            https://iss.moex.com/iss/securities.json?start=0&is_trading=1  
191            the argument should be {'start': 0, 'is_trading': '1'}
192                
193        Returns
194        -------
195        array_like
196            Returns the array of dictionaries from the MOEX ISS reply.
197            This array should be parsed and converted for usage.  
198            More information you can find on https://iss.moex.com/iss/reference/
199        """
200        _res = None
201        _values = self.base_values.copy()
202        try:
203            _mr = self.requests_dictionary[_type]
204            _url = self.base_url + _mr['postfix']
205            _rpp = _mr['postfix_params']
206            if _pparams:
207                for _ppkey in _pparams:
208                    if _ppkey in _rpp:
209                        _url = _url.replace(_ppkey, _pparams[_ppkey]) 
210            _rp = _mr['params']
211            if _params:
212                for _pp in _params:
213                    if _pp in _rp:
214                        _values[_pp] = f'{_params[_pp]:{_rp[_pp]:s}}'
215            _data = urllib.parse.urlencode(_values)
216            _url += f'?{_data:s}'
217            _req = urllib.request.Request(_url, headers=self.base_header, method = self.method)
218            _resp = None
219            try:
220                _resp = urllib.request.urlopen(_req, context=ssl._create_unverified_context())
221            except urllib.error.HTTPError as e:
222                print('_MoexRequest(): HTTP Error ', e.code)
223            except urllib.error.URLError as e:
224                print('_MoexRequest(): Error ', e.reason)
225            else:
226                _res = json.load(_resp)
227            if _resp:
228                _resp.close()
229        except Exception as e:
230            print('MoexImporter::_MoexRequest(): ', e, file=sys.stderr)
231        return _res

308    def _getSecurities(self, is_trading='', engine=None, market=None, query = None):
309        """Internal method to request security list.
310        
311        Parameters
312        ----------
313        is_trading: str  
314            Select traded, non-traded or all securities:  
315            ''  all securities,  
316            '0' - non-traded securities,  
317            '1' - traded securities.  
318        engine: str
319            Securities for the specific _engine.
320        market: str
321            Securities for the specific market.
322        query: str
323            Search security the part of its name or ticker.
324
325        Returns
326        -------
327        array_like
328            List of securities.
329        """
330        _res = None
331        _params = {
332            'start': 0,
333            'is_trading': is_trading,
334        }
335        _type = _MoexRequests.GetSecuritiesAll
336        if engine:
337            _type = _MoexRequests.GetSecuritiesForEngine
338            _params['engine'] = engine
339            if market:
340                _type = _MoexRequests.GetSecuritiesForMarket
341                _params['market'] = market
342        if query:
343            _type = _MoexRequests.GetSecuritiesSearch
344            _params['q'] = query
345        try:
346            isNext = True
347            while isNext:
348                isNext = False
349                _tmp = self._MoexRequest(
350                    _type,
351                    _params = _params
352                )
353                for _ti in _tmp:
354                    if 'securities' in _ti:
355                        if _ti['securities']:
356                            if not _res:
357                                _res = []
358                                
359                            _res += [
360                                {
361                                    _k:_sq[_k]
362                                    for _k in _sq
363                                    if _k in ['secid', 'shortname', 'name', 'regnumber', 'isin', 'is_traded', 'emitent_id', 'emitent_title', 'emitent_inn', 'gosreg', 'primary_boardid']
364                                } for _sq in _ti['securities']
365                            ]
366                            _params['start'] += self.limit
367                            if len(_ti['securities']) == self.limit:
368                                isNext = True
369                    
370        except Exception as e:
371            print('MoexImporter::_getSecurities(): ', e, file=sys.stderr)
372        return _res

160    def _MoexRequest(self, _type, _pparams = None, _params = None):
161        """Internal method for standartize https-queries to MOEX ISS.  
162        
163        You don't need to call this method directly, but you may use it for
164        more specific queries.  
165        
166        This method is wrapped into more direct methods to request data.
167        
168        Parameters
169        ----------
170        _type: _MoexRequests
171            Determines the type of the request. Is used to generate an url with
172            required parameters. Algorithms to generate the url are implemented
173            in __init__ method of the class.
174        _pparams: dict, optional
175            Dictionary of pairs id:value where id is the string id of the
176            parameter and value is its string value. These parameters are
177            used in the url string and should be replace with values.  
178            Example:  
179            in the url template  
180            https://iss.moex.com/iss/securities.json  
181            parameter __SECCODE__ should be replaced with the actual security
182            ticker.  {'__SECCODE___':'GAZP'} is passed to the method.
183        _params: dict, optional
184            Dictionary of pairs id:value where id is a string id of the
185            parameter and value is its value. The value type depends on the
186            request and automatically converted for the url in this method.  
187            Example:  
188            request  
189            https://iss.moex.com/iss/securities.json  
190            may use 2 parameters in some cases: start and is_trading  
191            https://iss.moex.com/iss/securities.json?start=0&is_trading=1  
192            the argument should be {'start': 0, 'is_trading': '1'}
193                
194        Returns
195        -------
196        array_like
197            Returns the array of dictionaries from the MOEX ISS reply.
198            This array should be parsed and converted for usage.  
199            More information you can find on https://iss.moex.com/iss/reference/
200        """
201        _res = None
202        _values = self.base_values.copy()
203        try:
204            _mr = self.requests_dictionary[_type]
205            _url = self.base_url + _mr['postfix']
206            _rpp = _mr['postfix_params']
207            if _pparams:
208                for _ppkey in _pparams:
209                    if _ppkey in _rpp:
210                        _url = _url.replace(_ppkey, _pparams[_ppkey]) 
211            _rp = _mr['params']
212            if _params:
213                for _pp in _params:
214                    if _pp in _rp:
215                        _values[_pp] = f'{_params[_pp]:{_rp[_pp]:s}}'
216            _data = urllib.parse.urlencode(_values)
217            _url += f'?{_data:s}'
218            _req = urllib.request.Request(_url, headers=self.base_header, method = self.method)
219            _resp = None
220            try:
221                _resp = urllib.request.urlopen(_req, context=ssl._create_unverified_context())
222            except urllib.error.HTTPError as e:
223                print('_MoexRequest(): HTTP Error ', e.code)
224            except urllib.error.URLError as e:
225                print('_MoexRequest(): Error ', e.reason)
226            else:
227                _res = json.load(_resp)
228            if _resp:
229                _resp.close()
230        except Exception as e:
231            print('MoexImporter::_MoexRequest(): ', e, file=sys.stderr)
232        return _res
