 14    def __init__(self, _seccode, _mi):
 15        """Class constructor initializes base variables
 16        and loads security-specific information from
 17        MOEX ISS.
 18        
 19        Parameters
 20        ----------
 21        _seccode: str
 22            Security ticker from MOEX.
 23        _mi: MoexImporter
 24            The object of MoexImporter that was
 25            created before. You can't use the class
 26            without this object.
 27        """
 28        self.seccode = _seccode
 29        self.shortname = None
 30        self.mainboard = None
 31        self.facecurrency = None
 32        self.facevalue = None
 33        self.mi = _mi
 34        self.boards = {}
 35        if isinstance(_mi, MoexImporter):
 36            _tmp = _mi.getSecurity(_seccode)
 37            for _ti in _tmp:
 38                if 'description' in _ti:
 39                    _sd = _ti['description']
 40                    for _si in _sd:
 41                        if _si['name'] == 'SHORTNAME':
 42                            self.shortname = _si['value']
 43                        if _si['name'] == 'FACEVALUE':
 44                            self.facevalue = _si['value']
 45                        if _si['name'] == 'FACEUNIT':
 46                            self.facecurrency = _si['value']
 47                if 'boards' in _ti:
 48                    _sb = _ti['boards']
 49                    for _bi in _sb:
 50                        _dtf = None
 51                        if _bi['history_from']:
 52                            _dtf = datetime.strptime(_bi['history_from'], '%Y-%m-%d').date()
 53                        _dtt = None
 54                        if _bi['history_till']:
 55                            _dtt = datetime.strptime(_bi['history_till'], '%Y-%m-%d').date()
 56                        self.boards[_bi['boardid']] = {
 57                            'dtfrom': _dtf,
 58                            'dttill': _dtt,
 59                            'engine': _bi['engine'],
 60                            'market': _bi['market'],
 61                            'title': _bi['title']
 62                        }
 63                        if _bi['is_primary'] == 1:
 64                            self.mainboard = _bi['boardid']
 65        else:
 66            print('MoexSecurity::__init__(): must be initialized with MoexImporter object.', file=sys.stderr)

 69        """Returns quotes for the security as a pandas dataframe.

 71        Parameters

 79            is used if the parameter is ommited. You can check

 81        _ts: MoexSessions, optional
 82            Request quotes for the specific session. The main session
 83            is used if the parameter is ommited.
 84
 85        Returns
 86        --------
 87        pd.DataFrame
 88            Quotes as pandas dataframe.
 89            Columns:
 90            'TRADEDATE' - date of the quote,
 91            'OPEN' - open price,
 92            'HIGH' - high price,
 93            'LOW' - low price,
 94            'CLOSE' - last price,
 95            'YIELD' - yield to maturity, may be None for non-bonds,
 96            'DURATION' - duration in days, may be None for non-bonds,
 97            'VALUE' - trading value in rubles,
 98            'QUANTITY' - trading value in securities.
 99        """
100        _res = None
101        try:
102            _tmp = self.getHistoryQuotesAsArray(_dtf=_dtf, _dtt=_dtt, _board=_board, _ts=_ts)
103            _res = pd.DataFrame.from_dict(data=_tmp, )
104            _res.set_index(['TRADEDATE',], inplace=True)
105            _res.sort_index(inplace=True)
106        except Exception as e:
107            print('MoexSecurity::getHistoryQuotesAsDataFrame(): ', e, file=sys.stderr)
108        return _res
